const openPanels = new Set();
function toggle(i){
  const demo=document.getElementById('demo-'+i);
  const tog=document.getElementById('toggle-'+i);
  if(openPanels.has(i)){openPanels.delete(i);demo.classList.remove('open');tog.classList.remove('open');}

}

async function callGemini(systemPrompt, userMsg){
  const res = await fetch(GEMINI_URL,{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({
      contents:[{role:"user",parts:[{text:systemPrompt+"\n\n"+userMsg}]}],
      generationConfig:{maxOutputTokens:1000,temperature:0.7}
    })
  });
  const data = await res.json();
  return data.candidates?.[0]?.content?.parts?.[0]?.text || "Error getting response.";
}

async function generateScript(){
  const name=document.getElementById('s-name').value.trim();
  const company=document.getElementById('s-company').value.trim();
  const role=document.getElementById('s-role').value.trim();
  const industry=document.getElementById('s-industry').value.trim();
  const pain=document.getElementById('s-pain').value.trim();
  if(!name||!company){alert("Please enter Prospect Name and Company.");return;}
  const btn=document.getElementById('sales-btn');
  const out=document.getElementById('sales-output');
  btn.disabled=true;btn.textContent="Generating...";
  out.style.display="block";
  out.innerHTML='<span class="typing"><span class="typing-dot"></span><span class="typing-dot"></span><span class="typing-dot"></span> Generating BANT script...</span>';
  const system=`You are an expert B2B sales AI agent trained on BANT qualification methodology. Generate a personalized outbound call script with: 1) Opening 2) Discovery (3 questions) 3) Tailored Pitch 4) Objection Handling (2 objections) 5) BANT Qualification Close. Use clear section headers. Keep it conversational.`;
  const user=`Generate BANT outbound call script for:\n- Prospect: ${name}, ${role||'Decision Maker'} at ${company}\n- Industry: ${industry||'SaaS'}\n- Pain point: ${pain||'scaling their sales team'}`;
  try{const resp=await callGemini(system,user);out.textContent=resp;}
  catch(e){out.textContent="Error. Please check your Gemini API key.";}
  btn.disabled=false;btn.textContent="Generate Call Script";
}

async function sendChat(){
  const inputEl=document.getElementById('chat-input');
  const msg=inputEl.value.trim();
  if(!msg)return;
  inputEl.value="";
  const context=document.getElementById('rag-context').value.trim();
  const emptyEl=document.getElementById('chat-empty');
  if(emptyEl)emptyEl.remove();
  appendMsg('user',msg);
  const system=context
    ?`You are an intelligent RAG-based document assistant. Answer questions ONLY based on this document context. If the answer is not in context, say so.\n\n--- DOCUMENT CONTEXT ---\n${context}\n--- END CONTEXT ---`
    :`You are a helpful AI assistant demonstrating a RAG chatbot. Mention the user can paste document content above for context-aware Q&A. Keep responses concise.`;
  const typingBubble=appendMsg('ai','<span class="typing"><span class="typing-dot"></span><span class="typing-dot"></span><span class="typing-dot"></span></span>',true);
  try{const resp=await callGemini(system,msg);typingBubble.textContent=resp;}
  catch(e){typingBubble.textContent="Error. Please check your Gemini API key.";}
}

function appendMsg(role,text,isHTML=false){
  const box=document.getElementById('chat-box');
  const wrap=document.createElement('div');wrap.className=`msg ${role}`;
  const avatar=document.createElement('div');avatar.className='msg-avatar';avatar.textContent=role==='ai'?'AI':'U';
  const bubble=document.createElement('div');bubble.className='msg-bubble';
  if(isHTML)bubble.innerHTML=text;else bubble.textContent=text;
  wrap.appendChild(avatar);wrap.appendChild(bubble);
  box.appendChild(wrap);box.scrollTop=box.scrollHeight;
  return bubble;
}
